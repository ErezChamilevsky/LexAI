User {
  _id: ObjectId,
  name: String,
  email: String,
  is_premium: Boolean,
  languages: [UserLanguage], 
  chats: [ObjectId],         // references Chat documents
  created_at: Date,
  updated_at: Date
}


UserLanguage {
  language_code: String,   // "en", "es", "fr", "he"
  level: String,           // "A1", "A2", "B1", "B2", "C1", etc.
  tests: [ObjectId]        // references Test documents
}


Language {
  _id: ObjectId,
  code: String,       // "en", "es", "he"
  name: String,       // "English", "Spanish", "Hebrew"
  created_at: Date
}


Test {
  _id: ObjectId,
  user_id: ObjectId,
  language_code: String,
  date_taken: Date,
  score: Number,         // % grade or projected CEFR level
  result_level: String,  // optional ("A1" -> "B1")
  details: Object,       // store breakdown of questions
}


Chat {
  _id: ObjectId,
  user_id: ObjectId,
  language_code: String,
  messages: [Message], 
  created_at: Date,
  updated_at: Date
}


Message {
  role: String,         // "user" or "assistant"
  content: String,
  timestamp: Date
}
